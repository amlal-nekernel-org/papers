\section{Abstract}
The note covers several formulas that have been written down for reusability purposes.

\section{Introduction and Preliminaries}
\begin{theorem}
There exists for $C(k)$:
\begin{equation}
C(k) \approx \sum_{p=1}^{n} \mathbb{I} (k \cdot p)
\end{equation}
Where $p > S(x) > n$. If and only if $x \geq 0$. Where $S(x)$ exists in $C(n ,p)$.
\end{theorem}

\section{On Theorems}
\begin{theorem}
There exists for $\mathbb{I}(t)$:
\begin{equation}
\mathbb{I}(t) = \int \frac{\mathbb{T}(t \cdot s)}{x^s}dx
\end{equation}
s.t. as an illustration, we can define $t$ as:
\begin{equation}
\mathbb{T}(s)=\sum_{n=1}^{\mathbb{L}} \frac{1}{n^{s}}
\end{equation}
Where $\mathbb{L}$ is a positive integer approaching $\pi$, $z$ is a limit approaching of a real approaching $\pi$ and $Log$ is the Principal Complex Logarithm of: $t \cdot z$, and $z \in \mathbb{C}$.
\end{theorem} {Per [1.1], and [1.2]:}
\begin{theorem}
$C(k) \in \mathbb{R}$ is a real number for all $n, p \in \mathbb{N}$ such that $p > S(x) > n$ and $x \geq 0$.
\end{theorem}
\begin{theorem}
$\mathbb{I}(t) \in \mathbb{R}$ is a real number for all $t > 0$ and $t \in \mathbb{R}$.
\end{theorem}

\section{Remarks}

The goal of this note was to establish the definitions of $C(k)$ and $\mathbb{I}(t)$, and to state the theorems that follow from these definitions. The properties of these functions can be further explored in future work.
