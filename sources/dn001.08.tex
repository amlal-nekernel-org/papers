\section{DEFINITIONS}

\subsection{DEFINITION OF A SUM OF $x \in \mathbb{R}$}
We define:
\begin{equation}
    S(x) \coloneqq \sum_{k=1}^{e}\frac{k \log(x)}{e}\cdot ke, \quad \int_{a}^{b}S(t)dt
\end{equation}
Where $\forall x \in \mathbb{R}, x > 1$.

\subsection{DEFINITION OF A SUM OF $(x \cdot e), \quad ((x+1)\cdot e) \in \mathbb{R}$}
We define:
\begin{equation}
    W(x) \coloneqq S(x \cdot e)\cdot S((x+1)\cdot e) +C, \quad \forall C \in \mathbb{Z}, \quad C > 0
\end{equation}
Where $\forall x \in \mathbb{R}, x > 0$.
